--  List out the employees who are earning salary between 3000 and 4500.

CREATE STREAM EMPLOYEE(
    employee_id     INT, 
    last_name       VARCHAR(30),
    first_name      VARCHAR(20),
    middle_name     CHAR(1),
    job_id          INT,
    manager_id      INT,
    hire_date       DATE,
    salary          FLOAT,
    commission      FLOAT,
    department_id   INT
    ) 
  FROM FILE 'examples/data/employee/employee.dat' LINE DELIMITED
  CSV ();

SELECT * 
FROM employee 
WHERE salary BETWEEN 2749 AND 4500
