--  Display employee details with salary grades.

CREATE STREAM SALARY_GRADE(
    grade_id     INT, 
    lower_bound  FLOAT,
    upper_bound  FLOAT
    ) 
  FROM FILE 'examples/data/employee/salary_grade.dat' LINE DELIMITED
  CSV ();

SELECT grade_id, lower_bound, upper_bound 
FROM salary_grade
