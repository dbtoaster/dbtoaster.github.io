-- How many employees in January month.

CREATE STREAM EMPLOYEE(
    employee_id     INT, 
    last_name       VARCHAR(30),
    first_name      VARCHAR(20),
    middle_name     CHAR(1),
    job_id          INT,
    manager_id      INT,
    hire_date       DATE,
    salary          FLOAT,
    commission      FLOAT,
    department_id   INT
    ) 
  FROM FILE 'examples/data/employee/employee.dat' LINE DELIMITED
  CSV ();

SELECT month(hire_date), count(*) 
FROM employee 
GROUP BY month(hire_date) 
HAVING month(hire_date)=1
