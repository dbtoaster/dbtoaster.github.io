  nPzL_7DkV.B AS B1, 
  nPzL_7DkV.C AS C1, 
  (nPzL_7DkV.B+(s1.B*s1.C)) AS LB71Y, 
  (nPzL_7DkV.C*nPzL_7DkV.B*(1*nPzL_7DkV.B)) AS gkkEoF, 
  S1.C AS qKKF7PYi

  (NOT 2 = nPzL_7DkV.C)
