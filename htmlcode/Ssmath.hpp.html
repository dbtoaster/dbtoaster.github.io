    void register_data(ProgramBase& pb) {
  
      pb.add_map<COUNT_map>( "COUNT", COUNT );
      pb.add_map<COUNT_mS1_map>( "COUNT_mS1", COUNT_mS1 );
      pb.add_map<COUNT_mS3_map>( "COUNT_mS3", COUNT_mS3 );

      pb.add_trigger("S", delete_tuple, std::bind(&data_t::unwrap_delete_S, this, std::placeholders::_1));

      exec_stats = pb.exec_stats;
      ivc_stats = pb.ivc_stats;

      exec_stats->register_probe(1, "insert_S_s1");
      exec_stats->register_probe(2, "insert_S_s2");
      exec_stats->register_probe(3, "delete_S_s0");
      exec_stats->register_probe(4, "delete_S_s1");
      exec_stats->register_probe(5, "delete_S_s2");
      #endif // DBT_PROFILE

    void on_insert_S(const long s_b, const long s_c) {
      BEGIN_TRIGGER(exec_stats,"insert_S")
      BEGIN_TRIGGER(ivc_stats,"insert_S")
      {  ++tN;
        long l1 = (s_c * (s_b * s_b));
        _c1.clear();
        { //foreach
          COUNT_mS1_entry* e1 = COUNT_mS1.head;

            add_to_temp_map(_c1, st1.modify(l2,result_3,v2));
            e1 = e1->nxt;
          }
        }
        long l3 = ((s_b * s_c) + s_b);
