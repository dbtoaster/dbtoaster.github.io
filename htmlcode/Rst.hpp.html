  struct tlq_t{
    struct timeval t0,t; long tT,tN,tS;

    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const {
  
      ar << "\n";
      const long _ATIMESD = get_ATIMESD();
      dbtoaster::serialize_nvp_tabbed(ar, STRING(ATIMESD), _ATIMESD, "\t");
  
    }

      return ATIMESD;
    
    }

    long ATIMESD;
  
  };

    /* Trigger functions for stream relations */
    void on_insert_R(const long r_a, const long r_b) {
      BEGIN_TRIGGER(exec_stats,"insert_R")
      BEGIN_TRIGGER(ivc_stats,"insert_R")
      {  ++tN;
        ATIMESD += (ATIMESD_mR1.getValueOrDefault(se2.modify(r_b)) * r_a);
        { //slice
          const HASH_RES_t h1 = ATIMESD_mT1_mR1_mapkey0_idxfn::hash(se4.modify0(r_b));
          const HashIndex_ATIMESD_mT1_mR1_map_0* i1 = static_cast<HashIndex_ATIMESD_mT1_mR1_map_0*>(ATIMESD_mT1_mR1.index[1]);
          HashIndex_ATIMESD_mT1_mR1_map_0::IdxNode* n1 = &(i1->buckets_[h1 % i1->size_]);
          ATIMESD_mT1_mR1_entry* e1;
          do if ((e1=n1->obj) && h1 == n1->hash && ATIMESD_mT1_mR1_mapkey0_idxfn::equals(se4, *e1)) {
            long atimesd_mtt_c = e1->ATIMESD_mTT_C;
            long v1 = e1->__av;
            ATIMESD_mT1.addOrDelOnZero(se3.modify(atimesd_mtt_c),(v1 * r_a));
          } while ((n1=n1->nxt));
        }
        ATIMESD_mS1.addOrDelOnZero(se5.modify(r_b),r_a);
      }
      END_TRIGGER(exec_stats,"insert_R")
      END_TRIGGER(ivc_stats,"insert_R")
    }
    void unwrap_insert_R(const event_args_t& ea) {
      on_insert_R(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<long*>(ea[1])));
