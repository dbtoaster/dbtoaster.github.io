  val diff_p = Math.pow(0.1,precision)
  
  def execute(args:Array[String],f:List[Any]=>Unit) = bench(args,(d:String,p:Int,t:Long,b:Int)=>run[Tpch6](Seq(

class Tpch6 extends Actor {
  import ddbt.lib.Messages._
  import Tpch6._
  import ddbt.lib.Functions._
  
  
  
  
  var REVENUE:Double = 0.0;
  
  def onAddLINEITEM(lineitem_orderkey:Long, lineitem_partkey:Long, lineitem_suppkey:Long, lineitem_linenumber:Long, lineitem_quantity:Double, lineitem_extendedprice:Double, lineitem_discount:Double, lineitem_tax:Double, lineitem_returnflag:String, lineitem_linestatus:String, lineitem_shipdate:Date, lineitem_commitdate:Date, lineitem_receiptdate:Date, lineitem_shipinstruct:String, lineitem_shipmode:String, lineitem_comment:String) {
    REVENUE += (if ((24L > lineitem_quantity)) {
      val x43 = if ((0.07 >= lineitem_discount)) {
        val x42 = if ((lineitem_discount >= 0.05)) {
          val x41 = if ((c2 > lineitem_shipdate)) {
            val x40 = if ((lineitem_shipdate >= c1)) {
              val x29 = lineitem_extendedprice * lineitem_discount
              x29
            } else {
              0L
            }
            x40
          } else {
            0L
          }
          x41
        } else {
          0L
        }
        x42
      } else {
        0L
      }
      x43
    } else {
      0L
    })
  }
  
  
  def onDelLINEITEM(lineitem_orderkey:Long, lineitem_partkey:Long, lineitem_suppkey:Long, lineitem_linenumber:Long, lineitem_quantity:Double, lineitem_extendedprice:Double, lineitem_discount:Double, lineitem_tax:Double, lineitem_returnflag:String, lineitem_linestatus:String, lineitem_shipdate:Date, lineitem_commitdate:Date, lineitem_receiptdate:Date, lineitem_shipinstruct:String, lineitem_shipmode:String, lineitem_comment:String) {
    val x27 = 24L > lineitem_quantity
    REVENUE += (if (x27) {
      val x25 = 0.07 >= lineitem_discount
      val x61 = if (x25) {
        val x23 = lineitem_discount >= 0.05
        val x60 = if (x23) {
          val x21 = c2 > lineitem_shipdate
          val x59 = if (x21) {
            val x18 = lineitem_shipdate >= c1
            val x58 = if (x18) {
              val x29 = lineitem_extendedprice * lineitem_discount
              val x47 = -x29
              x47
            } else {
              0L
            }
            x58
          } else {
            0L
          }
          x59
        } else {
          0L
        }
        x60
      } else {
        0L
      }
      x61
    } else {
      0L
    })
  }
  
  
  def onSystemReady() {
    REVENUE = 0.0
  }
  def printMapsInfo() = {}
  
  var t0=0L; var t1=0L; var tN=0L; var tS=0L
  def receive_skip:Receive = { case EndOfStream | GetSnapshot(_) => printMapsInfo(); sender ! (StreamStat(t1-t0,tN,tS),List(REVENUE)) case _ => tS+=1 }
  def receive = {
    case TupleEvent(ord,TupleInsert,"LINEITEM",List(v0:Long,v1:Long,v2:Long,v3:Long,v4:Double,v5:Double,v6:Double,v7:Double,v8:String,v9:String,v10:Date,v11:Date,v12:Date,v13:String,v14:String,v15:String)) => if (t1>0 && (tN&127)==0) { val t=System.nanoTime; if (t>t1) { t1=t; tS=1; context.become(receive_skip) } else tN+=1 } else tN+=1; onAddLINEITEM(v0,v1,v2,v3,v4,v5,v6,v7,v8,v9,v10,v11,v12,v13,v14,v15)
    case TupleEvent(ord,TupleDelete,"LINEITEM",List(v0:Long,v1:Long,v2:Long,v3:Long,v4:Double,v5:Double,v6:Double,v7:Double,v8:String,v9:String,v10:Date,v11:Date,v12:Date,v13:String,v14:String,v15:String)) => if (t1>0 && (tN&127)==0) { val t=System.nanoTime; if (t>t1) { t1=t; tS=1; context.become(receive_skip) } else tN+=1 } else tN+=1; onDelLINEITEM(v0,v1,v2,v3,v4,v5,v6,v7,v8,v9,v10,v11,v12,v13,v14,v15)
    case StreamInit(timeout) => onSystemReady(); t0=System.nanoTime; if (timeout>0) t1=t0+timeout*1000000L
    case EndOfStream | GetSnapshot(_) => t1=System.nanoTime; printMapsInfo(); sender ! (StreamStat(t1-t0,tN,tS),List(REVENUE))
  }
  val c1:Date = Udate("1994-1-1")
  val c2:Date = Udate("1995-1-1")
}
