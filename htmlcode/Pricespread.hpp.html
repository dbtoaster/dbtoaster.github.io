  struct tlq_t{

  /* Serialization Code */
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const {
  
      ar << "\n";
      const DOUBLE_TYPE _PSP = get_PSP();
      dbtoaster::serialize_nvp_tabbed(ar, STRING(PSP), _PSP, "\t");
  
    }
  
    /* Functions returning / computing the results of top level queries */
    const DOUBLE_TYPE get_PSP() const {
      return PSP;
    
    }
  
  protected:
  
    /* Data structures used for storing / computing top level queries */
    DOUBLE_TYPE PSP;
  
  };

  struct data_t : tlq_t{

    /* Registering relations and trigger functions */
    void register_data(ProgramBase& pb) {
  
      pb.add_map<DOUBLE_TYPE>( "PSP", PSP );
      pb.add_map<PSP_mBIDS1_map>( "PSP_mBIDS1", PSP_mBIDS1 );
      pb.add_map<DOUBLE_TYPE>( "PSP_mBIDS1_L1_1", PSP_mBIDS1_L1_1 );
      pb.add_map<PSP_mBIDS2_map>( "PSP_mBIDS2", PSP_mBIDS2 );

      pb.add_map<PSP_mBIDS3_map>( "PSP_mBIDS3", PSP_mBIDS3 );
      pb.add_map<PSP_mBIDS4_map>( "PSP_mBIDS4", PSP_mBIDS4 );
  
      pb.add_relation("BIDS");
      pb.add_relation("ASKS");
  
      
  
      pb.add_trigger("BIDS", insert_tuple, std::bind(&data_t::unwrap_insert_BIDS, this, std::placeholders::_1));
      pb.add_trigger("BIDS", delete_tuple, std::bind(&data_t::unwrap_delete_BIDS, this, std::placeholders::_1));
      pb.add_trigger("ASKS", insert_tuple, std::bind(&data_t::unwrap_insert_ASKS, this, std::placeholders::_1));
      pb.add_trigger("ASKS", delete_tuple, std::bind(&data_t::unwrap_delete_ASKS, this, std::placeholders::_1));
  
      #ifdef DBT_PROFILE
      exec_stats = pb.exec_stats;
      ivc_stats = pb.ivc_stats;
      exec_stats->register_probe(0, "insert_BIDS_s0");
      exec_stats->register_probe(1, "insert_BIDS_s1");

    void on_insert_BIDS(const DOUBLE_TYPE bids_t, const long bids_id, const long bids_broker_id, const DOUBLE_TYPE bids_volume, const DOUBLE_TYPE bids_price) {
      BEGIN_TRIGGER(exec_stats,"insert_BIDS")
      BEGIN_TRIGGER(ivc_stats,"insert_BIDS")
      {  ++tN;
        PSP_mBIDS2.addOrDelOnZero(se1.modify(bids_volume),1L);
        PSP_mBIDS2_L1_1 += bids_volume;
        PSP_mBIDS4.addOrDelOnZero(se3.modify(bids_volume),bids_price);
        DOUBLE_TYPE agg1 = 0.0;
        DOUBLE_TYPE l1 = (PSP_mBIDS1_L1_1 * 0.0001);
        { //foreach
          PSP_mBIDS1_entry* e1 = PSP_mBIDS1.head;
          while(e1){
            DOUBLE_TYPE a_volume = e1->A_VOLUME;
            DOUBLE_TYPE v1 = e1->__av;
            (/*if */(a_volume > l1) ? agg1 += v1 : 0.0);
            e1 = e1->nxt;
          }
        }
        long agg2 = 0L;
        DOUBLE_TYPE l2 = (PSP_mBIDS2_L1_1 * 0.0001);
        { //foreach
          PSP_mBIDS2_entry* e2 = PSP_mBIDS2.head;
          while(e2){
            DOUBLE_TYPE b_volume = e2->B_VOLUME;
            long v2 = e2->__av;
            (/*if */(b_volume > l2) ? agg2 += v2 : 0L);
            e2 = e2->nxt;
          }
        }
        long agg3 = 0L;
        DOUBLE_TYPE l3 = (PSP_mBIDS1_L1_1 * 0.0001);
        { //foreach
          PSP_mBIDS3_entry* e3 = PSP_mBIDS3.head;
          while(e3){
            DOUBLE_TYPE a_volume = e3->A_VOLUME;
            long v3 = e3->__av;
            (/*if */(a_volume > l3) ? agg3 += v3 : 0L);
            e3 = e3->nxt;
          }
        }
        DOUBLE_TYPE agg4 = 0.0;
        DOUBLE_TYPE l4 = (PSP_mBIDS2_L1_1 * 0.0001);
        { //foreach
          PSP_mBIDS4_entry* e4 = PSP_mBIDS4.head;
          while(e4){
            DOUBLE_TYPE b_volume = e4->B_VOLUME;
            DOUBLE_TYPE v4 = e4->__av;
            (/*if */(b_volume > l4) ? agg4 += v4 : 0.0);
            e4 = e4->nxt;
          }
        }
        PSP = ((agg1 * agg2) + (agg3 * (agg4 * -1L)));
      }
      END_TRIGGER(exec_stats,"insert_BIDS")
      END_TRIGGER(ivc_stats,"insert_BIDS")
    }
    void unwrap_insert_BIDS(const event_args_t& ea) {
      on_insert_BIDS(*(reinterpret_cast<DOUBLE_TYPE*>(ea[0])), *(reinterpret_cast<long*>(ea[1])), *(reinterpret_cast<long*>(ea[2])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[3])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[4])));
    }
